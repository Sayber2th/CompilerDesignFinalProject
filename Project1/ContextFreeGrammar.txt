Program → Statement Program | ε

Statement → "int" Identifier Initializer ";"
          | "string" Identifier Initializer ";"
          | Identifier "=" InitValue ";"
          | "print" "(" InitValue ")" ";"

Initializer → "=" InitValue | ε

InitValue → StringLiteral | Expression

Expression → Term Expression'
Expression' → "+" Term Expression'
            | "-" Term Expression'
            | ε

Term → Factor Term'
Term' → "*" Factor Term'
      | "/" Factor Term'
      | ε

Factor → Identifier
       | IntegerLiteral
       | "(" Expression ")"
